rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to perform operations on chat and messaging data
    match /chatRooms/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
    
    match /messages/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
    
    // Booking data - authenticated users only
    match /bookings/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
    
    // Reviews - authenticated users can read and write
    match /reviews/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
    
    // User data - authenticated users only
    match /users/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
    
    // Salon data - public read, authenticated write
    match /salons/{document=**} {
      allow read: if true;
      allow write: if request.auth.uid != null;
    }
    
    // Staff data - authenticated access
    match /staff/{document=**} {
      allow read, write: if request.auth.uid != null;
    }
  }
}
